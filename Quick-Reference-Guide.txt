Run the recursive command while in the c-dat.github.io folder in shell:
```
find . -name index.md -type f -delete
# deletes all the index files in all the sub-foldersin the tree
for x in $(find . -type d -print0)
do
    cp navigation.sh $x;
done;
# Overwrites the navigation.sh file into all the sub-folders in the tree
find . -name 'navigation.sh' -execdir ./navigation.sh \;
# creates the index files in all the sub-folders
find . -name navigation.sh -type f -delete
# deletes all the navigation.sh files in all the sub-folders in the tree
```
