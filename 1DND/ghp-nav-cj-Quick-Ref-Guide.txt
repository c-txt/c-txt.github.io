# Run the recursive command while in the c-dat.github.io folder in shell:
find . -name index.md -type f -delete
# deletes all the index files in all the sub-foldersin the tree
for x in $(find . -type d -print0  -name \.git -prune)
do
    cp ghp-nav-cj.sh $x;
done;
# Overwrites the ghp-nav-cj.sh file into all the sub-folders in the tree except the git folder
find . -name 'ghp-nav-cj.sh' -execdir ./ghp-nav-cj.sh \;
# creates the index files in all the sub-folders
find . -name ghp-nav-cj.sh -type f -delete
# deletes all the ghp-nav-cj.sh files in all the sub-folders in the tree
sed -iE '/Up to Parent folder/d' index.md
# removes the line containing the above pattern in the index file in the root folder
rm index.mdE
# deletes the backup file created by the above sed command
